create table t_user(
    id           bigint auto_increment comment 'pk id'
        primary key,
    username     varchar(256)           not null,
    real_name    varchar(32) default '' not null,
    nick_name    varchar(32) default '' not null,
    password     varchar(256)           not null,
    phone_number varchar(16) default '' not null,
    locale       varchar(8)  default '' not null,
    country      varchar(2)  default '' not null
) comment 'user table';

create index idx_country_locale_user
    on t_user (country, locale, username);

create index idx_locale_user
    on t_user (locale, username);

create index idx_real_name
    on t_user (real_name);

select * from mysql.innodb_index_stats 
where table_name = 't_user';
