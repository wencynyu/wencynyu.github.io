                            readBuffer.clear();

                            // Êìç‰Ωú+rÔºåÈÄöËøáÊåâ‰ΩçÊàñÊù•ËøõË°å‰øÆÊîπ
                            key.interestOps(key.interestOps() | SelectionKey.OP_WRITE);
                        }

                            int len = 0;
                            while(writeBuffer.hasRemaining()){
                                len = sc.write(writeBuffer);
                                /*ËØ¥ÊòéÂ∫ïÂ±ÇÁöÑsocketÂÜôÁºìÂÜ≤Â∑≤Êª°*/
                                if(len == 0){

                            if(len != 0){
                                /*ÂèñÊ∂àÈÄöÈÅìÁöÑÂÜô‰∫ã‰ª∂*/
                                key.interestOps(key.interestOps() & (~SelectionKey.OP_WRITE));
                            }
                        }
                    }catch (Exception e){
                        System.out.println("service encounter client error");
                        e.printStackTrace();
                        // ÂÆ¢Êà∑Á´ØËøûÊé•Âá∫Áé∞ÂºÇÂ∏∏Ôºå‰ªéselector‰∏≠ÁßªÈô§Ëøô‰∏™key
                        key.cancel();
                        key.channel().close();
                    }

                }
            }
        }catch (Exception e){
            System.out.println("serverThread is interrupted");
        }finally {
            try{
                selector.close();
            }catch(Exception e){
                System.out.println("selector close failed");
            }finally{
                System.out.println("server close");
            }
        }

    }
}

public class NettyServer {
    public static void main(String[] args) {
        EventLoopGroup bossGroup = new NioEventLoopGroup(1);
        EventLoopGroup workerGroup = new NioEventLoopGroup(Runtime.getRuntime().availableProcessors() * 2);
        try {
            ServerBootstrap bootstrap = new ServerBootstrap();
            bootstrap.group(bossGroup, workerGroup).channel(NioServerSocketChannel.class)
                    .localAddress(new InetSocketAddress(8000))
                    .childHandler(new ChannelInitializer<SocketChannel>() {
                        @Override
                        protected void initChannel(SocketChannel socketChannel) throws Exception {
                            socketChannel.pipeline().addLast(new NettyServerHandler());
                        }
                    });
            ChannelFuture future = bootstrap.bind().sync();
            future.channel().closeFuture().sync();
        } catch (Exception e) {
            e.printStackTrace();
        } finally {
            try {
                bossGroup.shutdownGracefully().sync();
                workerGroup.shutdownGracefully().sync();
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
        }
    }
}
